-- ============================================================ -- ระบบรายงานรายสัปดาห์ — ตั้งค่าฐานข้อมูล Supabase -- วิธีใช้: เปิดโปรเจกต์ Supabase > เมนู SQL Editor > New query -- วางทั้งหมดนี้ > กด Run (ครั้งเดียวจบ) -- ============================================================ -- เข้ารหัสรหัสผ่าน (bcrypt) create extension if not exists pgcrypto; -- ---------- ตาราง ---------- create table if not exists app_users ( username text primary key, password_hash text not null, display_name text, role text default 'user', token text ); create table if not exists reports ( id bigint generated always as identity primary key, username text not null references app_users(username) on delete cascade, week_key text not null, range_label text, data jsonb, updated_at timestamptz default now() ); create index if not exists idx_reports_user on reports(username); -- ปิดการเข้าถึงตารางตรงๆ จาก anon (ต้องผ่านฟังก์ชันเท่านั้น) alter table app_users enable row level security; alter table reports enable row level security; -- ---------- helper ---------- create or replace function app_role(p_username text, p_role text) returns text language sql immutable as $$ select case when lower(coalesce(p_role,'')) = 'admin' or lower(trim(p_username)) = 'admin' then 'admin' else 'user' end; $$; -- คืน "ชุดแถว" (setof) เพื่อให้เมื่อ token ไม่ตรง = ได้ 0 แถว (NOT FOUND) จริงๆ -- ถ้าใช้ returns app_users เฉยๆ เมื่อไม่ตรงจะได้ 1 แถวที่เป็น null ทั้งหมด (บั๊ก) drop function if exists app_auth(text,text); create or replace function app_auth(p_username text, p_token text) returns setof app_users language sql stable as $$ select * from app_users where username = trim(p_username) and token is not null and token = encode(digest(p_token, 'sha256'), 'hex'); -- เทียบกับ hash ของ token $$; -- ---------- auth ---------- create or replace function app_login(p_username text, p_password text) returns jsonb language plpgsql security definer as $$ declare u app_users; t text; begin select * into u from app_users where username = trim(p_username); if not found or u.password_hash <> crypt(p_password, u.password_hash) then return jsonb_build_object('ok', false, 'error', 'ชื่อผู้ใช้หรือรหัสผ่านไม่ถูกต้อง'); end if; t := encode(gen_random_bytes(16), 'hex'); -- token จริง (ส่งให้เว็บ) update app_users set token = encode(digest(t, 'sha256'), 'hex') -- เก็บเฉพาะ hash ในฐานข้อมูล where username = u.username; -- session ใหม่ = เตะเครื่องเก่า return jsonb_build_object('ok', true, 'displayName', coalesce(u.display_name, u.username), 'role', app_role(u.username, u.role), 'token', t); end; $$; create or replace function app_ping(p_username text, p_token text) returns jsonb language plpgsql security definer as $$ declare u app_users; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; return jsonb_build_object('ok', true, 'displayName', coalesce(u.display_name, u.username), 'role', app_role(u.username, u.role)); end; $$; -- ---------- รายงาน ---------- -- บันทึกแบบ 1 รายงานต่อสัปดาห์ (ทับของเดิม + เก็บความซ้ำเก่าออก) create or replace function app_save(p_username text, p_token text, p_week_key text, p_range_label text, p_data jsonb) returns jsonb language plpgsql security definer as $$ declare u app_users; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; delete from reports where username = u.username and week_key = p_week_key; -- ลบของสัปดาห์นี้ทิ้งก่อน (รวมของซ้ำเก่า) insert into reports(username, week_key, range_label, data, updated_at) values (u.username, p_week_key, p_range_label, p_data, now()); return jsonb_build_object('ok', true); end; $$; -- โหลดเวอร์ชันล่าสุดของสัปดาห์ create or replace function app_load(p_username text, p_token text, p_week_key text) returns jsonb language plpgsql security definer as $$ declare u app_users; r reports; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; select * into r from reports where username = u.username and week_key = p_week_key order by updated_at desc limit 1; if not found then return jsonb_build_object('ok', true, 'data', null); end if; return jsonb_build_object('ok', true, 'data', r.data, 'rangeLabel', r.range_label); end; $$; -- โหลดเวอร์ชันเจาะจงตาม id (admin เปิดของใครก็ได้ / ผู้ใช้ทั่วไปเปิดได้เฉพาะของตัวเอง) create or replace function app_load_row(p_username text, p_token text, p_row bigint) returns jsonb language plpgsql security definer as $$ declare u app_users; r reports; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; if app_role(u.username, u.role) = 'admin' then select * into r from reports where id = p_row; else select * into r from reports where id = p_row and username = u.username; end if; if not found then return jsonb_build_object('ok', false, 'error', 'ไม่พบรายการ'); end if; return jsonb_build_object('ok', true, 'data', r.data, 'weekKey', r.week_key, 'rangeLabel', r.range_label, 'owner', r.username); end; $$; -- admin: รายการรายงานของทุกคน (ใหม่สุดก่อน) create or replace function app_weeks_all(p_username text, p_token text) returns jsonb language plpgsql security definer as $$ declare u app_users; res jsonb; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; if app_role(u.username, u.role) <> 'admin' then return jsonb_build_object('ok', false, 'error', 'เฉพาะผู้ดูแลระบบ'); end if; select coalesce(jsonb_agg(jsonb_build_object( 'row', id, 'owner', username, 'weekKey', week_key, 'rangeLabel', range_label, 'updatedAt', updated_at) order by updated_at desc), '[]'::jsonb) into res from reports; return jsonb_build_object('ok', true, 'weeks', res); end; $$; -- รายการย้อนหลังทั้งหมด (ใหม่สุดก่อน) create or replace function app_weeks(p_username text, p_token text) returns jsonb language plpgsql security definer as $$ declare u app_users; res jsonb; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; select coalesce(jsonb_agg(jsonb_build_object( 'row', id, 'weekKey', week_key, 'rangeLabel', range_label, 'updatedAt', updated_at) order by updated_at desc), '[]'::jsonb) into res from reports where username = u.username; return jsonb_build_object('ok', true, 'weeks', res); end; $$; create or replace function app_delete_week(p_username text, p_token text, p_row bigint) returns jsonb language plpgsql security definer as $$ declare u app_users; n int; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; delete from reports where id = p_row and username = u.username; get diagnostics n = row_count; if n = 0 then return jsonb_build_object('ok', false, 'error', 'ไม่พบรายการ'); end if; return jsonb_build_object('ok', true); end; $$; -- ---------- admin: จัดการผู้ใช้ ---------- create or replace function app_add_user(p_username text, p_token text, p_new_username text, p_new_password text, p_display text, p_role text) returns jsonb language plpgsql security definer as $$ declare u app_users; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; if app_role(u.username, u.role) <> 'admin' then return jsonb_build_object('ok', false, 'error', 'เฉพาะผู้ดูแลระบบ'); end if; if trim(coalesce(p_new_username,'')) = '' or coalesce(p_new_password,'') = '' then return jsonb_build_object('ok', false, 'error', 'กรอกชื่อผู้ใช้และรหัสผ่าน'); end if; if exists(select 1 from app_users where username = trim(p_new_username)) then return jsonb_build_object('ok', false, 'error', 'มีชื่อผู้ใช้นี้แล้ว'); end if; insert into app_users(username, password_hash, display_name, role, token) values (trim(p_new_username), crypt(p_new_password, gen_salt('bf')), coalesce(p_display, p_new_username), case when p_role = 'admin' then 'admin' else 'user' end, null); return jsonb_build_object('ok', true); end; $$; create or replace function app_list_users(p_username text, p_token text) returns jsonb language plpgsql security definer as $$ declare u app_users; res jsonb; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; if app_role(u.username, u.role) <> 'admin' then return jsonb_build_object('ok', false, 'error', 'เฉพาะผู้ดูแลระบบ'); end if; select coalesce(jsonb_agg(jsonb_build_object( 'username', username, 'displayName', display_name, 'role', app_role(username, role)) order by username), '[]'::jsonb) into res from app_users; return jsonb_build_object('ok', true, 'users', res); end; $$; create or replace function app_delete_user(p_username text, p_token text, p_target text) returns jsonb language plpgsql security definer as $$ declare u app_users; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; if app_role(u.username, u.role) <> 'admin' then return jsonb_build_object('ok', false, 'error', 'เฉพาะผู้ดูแลระบบ'); end if; if lower(trim(p_target)) = 'admin' then return jsonb_build_object('ok', false, 'error', 'ลบบัญชี admin ไม่ได้'); end if; if trim(p_target) = u.username then return jsonb_build_object('ok', false, 'error', 'ลบบัญชีตัวเองไม่ได้'); end if; delete from app_users where username = trim(p_target); return jsonb_build_object('ok', true); end; $$; -- ผู้ใช้เปลี่ยนรหัสผ่านของตัวเอง (ต้องใส่รหัสเดิมให้ถูก) create or replace function app_change_password(p_username text, p_token text, p_old_password text, p_new_password text) returns jsonb language plpgsql security definer as $$ declare u app_users; rec app_users; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; select * into rec from app_users where username = u.username; if rec.password_hash <> crypt(p_old_password, rec.password_hash) then return jsonb_build_object('ok', false, 'error', 'รหัสผ่านปัจจุบันไม่ถูกต้อง'); end if; if length(coalesce(p_new_password,'')) < 4 then return jsonb_build_object('ok', false, 'error', 'รหัสผ่านใหม่สั้นเกินไป (อย่างน้อย 4 ตัว)'); end if; update app_users set password_hash = crypt(p_new_password, gen_salt('bf')) where username = u.username; return jsonb_build_object('ok', true); end; $$; -- admin รีเซ็ตรหัสผ่านให้ผู้ใช้คนอื่น (คนนั้นจะถูกบังคับล็อกอินใหม่) create or replace function app_reset_password(p_username text, p_token text, p_target text, p_new_password text) returns jsonb language plpgsql security definer as $$ declare u app_users; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; if app_role(u.username, u.role) <> 'admin' then return jsonb_build_object('ok', false, 'error', 'เฉพาะผู้ดูแลระบบ'); end if; if not exists(select 1 from app_users where username = trim(p_target)) then return jsonb_build_object('ok', false, 'error', 'ไม่พบผู้ใช้'); end if; if length(coalesce(p_new_password,'')) < 4 then return jsonb_build_object('ok', false, 'error', 'รหัสผ่านสั้นเกินไป (อย่างน้อย 4 ตัว)'); end if; update app_users set password_hash = crypt(p_new_password, gen_salt('bf')), token = null where username = trim(p_target); return jsonb_build_object('ok', true); end; $$; -- ส่งออกข้อมูลทั้งหมด (admin เท่านั้น) — ใช้กับ auto-backup create or replace function app_export_all(p_username text, p_token text) returns jsonb language plpgsql security definer as $$ declare u app_users; begin select * into u from app_auth(p_username, p_token); if not found then return jsonb_build_object('ok', false, 'error', 'session'); end if; if app_role(u.username, u.role) <> 'admin' then return jsonb_build_object('ok', false, 'error', 'เฉพาะผู้ดูแลระบบ'); end if; return jsonb_build_object( 'ok', true, 'exported_at', now(), 'count', (select count(*) from reports), 'users', (select coalesce(jsonb_agg(jsonb_build_object( 'username', username, 'displayName', display_name, 'role', role) order by username), '[]'::jsonb) from app_users), 'reports', (select coalesce(jsonb_agg(jsonb_build_object( 'id', id, 'username', username, 'week_key', week_key, 'range_label', range_label, 'data', data, 'updated_at', updated_at) order by username, updated_at), '[]'::jsonb) from reports) ); end; $$; -- ---------- สิทธิ์: ให้ anon เรียกได้เฉพาะฟังก์ชันเหล่านี้ ---------- grant execute on function app_login(text,text), app_ping(text,text), app_save(text,text,text,text,jsonb), app_load(text,text,text), app_load_row(text,text,bigint), app_weeks(text,text), app_delete_week(text,text,bigint), app_add_user(text,text,text,text,text,text), app_list_users(text,text), app_delete_user(text,text,text), app_change_password(text,text,text,text), app_reset_password(text,text,text,text), app_export_all(text,text), app_weeks_all(text,text) to anon, authenticated;
